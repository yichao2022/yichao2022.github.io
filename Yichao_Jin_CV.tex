\section*{Working Papers}
\begin{enumerate}
    \item \textbf{``Waiting Time as a Behavioral Barrier to Vaccination Uptake: Nonlinear Heterogeneity by Institutional Trust''} \\
    \textit{Job Market Paper} --- \href{https://yichao2022.github.io/jmp.pdf}{\textbf{Link to Full Draft}}
    \begin{itemize}[noitemsep, topsep=0pt]
        \item Developed a structural Discrete Choice Experiment (DCE) framework to analyze the interaction between temporal barriers and institutional trust in vaccination behavior.
        \item Identified nonlinear heterogeneity in waiting time sensitivity, providing a behavioral explanation for vaccine hesitancy.
    \end{itemize}

    \item \textbf{``Behavioral Digital Twins: A Causally Constrained Generative Framework for Synthetic Public Health Policy Simulation''} \\
    \textit{Working Paper} --- Submitted to DAISY 2026 Workshop @ ACM BCB. Developed a 3-stage framework integrating DCE data and Structural Causal Modeling (SCM) to align generative agents with empirical choice frontiers, achieving a 34\% reduction in frontier-deviation MSE.

    \item \textbf{``The Ghost in the Machine: Algorithmic Rationality vs. Social Legitimacy in AI-Driven Public Health Governance''} \\
    \textit{Working Paper} --- Analyzes the theoretical tension between automated efficiency and institutional trust using the AGIL framework and Rational Choice models.

    \item \textbf{``The Price of Waiting: Evidence on Cash--Time Trade-Offs in Vaccination Time Discount''} \\
    \textit{Manuscript in Preparation (Expected submission: August 2026)} --- Targeting \textit{Social Science \& Medicine}.
\end{enumerate}
